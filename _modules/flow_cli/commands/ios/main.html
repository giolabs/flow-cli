def show_ios_menu(ctx: click.Context) -> None:
    """Show interactive iOS menu"""
    import inquirer
    from flow_cli.core.ui.banner import show_section_header
    
    show_section_header("iOS Development Tools", "üçé")
    
    choices = [
        "üèóÔ∏è  Build - Build iOS app",
        "‚ñ∂Ô∏è  Run - Run on iOS simulator/device",
        "üì± Install - Install IPA on devices",
        "üîå Devices - Manage iOS devices and simulators",
        "üé® Flavors - View available flavors",
        "üîô Back to main menu"
    ]
    
    try:
        questions = [
            inquirer.List(
                'action',
                message="Select iOS action:",
                choices=choices,
                carousel=True
            ),
        ]
        
        answers = inquirer.prompt(questions)
        if not answers:
            return
            
        action = answers['action']
        
        if action.startswith("üèóÔ∏è"):
            console.print("[yellow]üöß iOS build command coming soon![/yellow]")
            console.print("This will include:")
            console.print("  ‚Ä¢ Build for iOS simulator")
            console.print("  ‚Ä¢ Build for physical device")
            console.print("  ‚Ä¢ Archive for App Store")
            console.print("  ‚Ä¢ Flavor-specific builds")
        elif action.startswith("‚ñ∂Ô∏è"):
            ctx.invoke(run_command)
        elif action.startswith("üì±"):
            console.print("[yellow]üöß iOS install command coming soon![/yellow]")
            console.print("This will include:")
            console.print("  ‚Ä¢ Install IPA on connected devices")
            console.print("  ‚Ä¢ Install on simulators")
            console.print("  ‚Ä¢ TestFlight integration")
        elif action.startswith("üîå"):
            ctx.invoke(devices_command)
        elif action.startswith("üé®"):
            ctx.invoke(flavors_command)
        elif action.startswith("üîô"):
            return
            
    except KeyboardInterrupt:
        console.print("\n[dim]Returning to main menu...[/dim]")
        return
