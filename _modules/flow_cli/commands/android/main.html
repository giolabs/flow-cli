def show_android_menu(ctx: click.Context) -> None:
    """Show interactive Android menu"""
    import inquirer
    from rich.console import Console

    from flow_cli.core.ui.banner import show_section_header

    console = Console()
    show_section_header("Android Development Tools", "ü§ñ")

    choices = [
        "üèóÔ∏è  Build - Build Android APK/AAB",
        "‚ñ∂Ô∏è  Run - Run on Android device/emulator",
        "üì± Install - Install APK on devices",
        "üîå Devices - Manage Android devices",
        "üé® Flavors - View available flavors",
        "üîô Back to main menu",
    ]

    try:
        questions = [
            inquirer.List(
                "action", message="Select Android action:", choices=choices, carousel=True
            ),
        ]

        answers = inquirer.prompt(questions)
        if not answers:
            return

        action = answers["action"]

        if action.startswith("üèóÔ∏è"):
            ctx.invoke(build_command)
        elif action.startswith("‚ñ∂Ô∏è"):
            ctx.invoke(run_command)
        elif action.startswith("üì±"):
            ctx.invoke(install_command)
        elif action.startswith("üîå"):
            ctx.invoke(devices_command)
        elif action.startswith("üé®"):
            ctx.invoke(flavors_command)
        elif action.startswith("üîô"):
            return

    except KeyboardInterrupt:
        console.print("\n[dim]Returning to main menu...[/dim]")
        return
